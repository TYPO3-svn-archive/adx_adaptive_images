
/**
 * plugin.tx_adxadaptiveimages : default settings
 */
plugin.tx_adxadaptiveimages {

	cssClassPrefix = {$plugin.tx_adxadaptiveimages.cssClassPrefix}

	breakpoints {

		0 {

			ratio = {$plugin.tx_adxadaptiveimages.breakpoints.0.ratio}

			image < tt_content.image.20.1
			image.file.maxW = {$plugin.tx_adxadaptiveimages.breakpoints.0.maxWidth}
		}

		1 < .0
		1 {
			ratio = {$plugin.tx_adxadaptiveimages.breakpoints.1.ratio}
			image.file.maxW = {$plugin.tx_adxadaptiveimages.breakpoints.1.maxWidth}
		}

		2 < .0
		2 {
			ratio = {$plugin.tx_adxadaptiveimages.breakpoints.2.ratio}
			image.file.maxW = {$plugin.tx_adxadaptiveimages.breakpoints.2.maxWidth}
		}

		noScript < .0
	}
}


// add less variable
plugin.tx_adxless {

	variables {
		adxadaptiveimagesClassPrefix = {$plugin.tx_adxadaptiveimages.cssClassPrefix}
	}
}

// make tt_content.image adaptive
tt_content.image {
	20.1.adaptive < plugin.tx_adxadaptiveimages
	20.1.adaptive.enable.field = tx_adxadaptiveimages_adaptive
}

// include CSS
page.includeCSS {
	tx_adxadaptiveimages_css = EXT:adx_adaptive_images/Resources/Private/CSS/Styles.adxless.css
}

// include JavaScript
page.includeJS {
	tx_adxadaptiveimages_js = EXT:adx_adaptive_images/Resources/Public/JavaScript/AdaptiveImages.js
}
