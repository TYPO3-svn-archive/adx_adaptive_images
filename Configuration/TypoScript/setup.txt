
/**
 * plugin.tx_adxadaptiveimages
 */
plugin.tx_adxadaptiveimages {

	cssClassPrefix = {$plugin.tx_adxadaptiveimages.cssClassPrefix}

	breakpoints {

		noScript < tt_content.image.20.1
		noScript {
			file.maxW = {$plugin.tx_adxadaptiveimages.maxWidth.default}
		}

		0 < tt_content.image.20.1
		0 {
			file.maxW = {$plugin.tx_adxadaptiveimages.maxWidth.large}
		}

		1 < tt_content.image.20.1
		1 {
			file.maxW = {$plugin.tx_adxadaptiveimages.maxWidth.default}
		}

		2 < tt_content.image.20.1
		2 {
			file.maxW = {$plugin.tx_adxadaptiveimages.maxWidth.medium}
		}
	}

}

/**
 * plugin.tx_adxless
 */
plugin.tx_adxless {

	variables {
		adxadaptiveimagesClassPrefix = {$plugin.tx_adxadaptiveimages.cssClassPrefix}
	}
